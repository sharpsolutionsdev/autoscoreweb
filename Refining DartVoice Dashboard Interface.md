# Open the image
img = Image.open('image copy.png').convert('RGBA')
width, height = img.size

# Background color we want to remove
bg_color = (28, 31, 36)

# Create a new image for the output
new_img = Image.new('RGBA', img.size)

# Tolerance for anti-aliasing
tolerance = 30

bbox = [width, height, 0, 0] # min_x, min_y, max_x, max_y

pixels = img.load()
new_pixels = new_img.load()

for y in range(height):
    for x in range(width):
        r, g, b, a = pixels[x, y]
        
        # Calculate distance to bg_color
        dist = ((r - bg_color[0])**2 + (g - bg_color[1])**2 + (b - bg_color[2])**2)**0.5
        
        if dist < tolerance:
            # Make fully transparent if exactly bg, or partially transparent for edges
            alpha = int((dist / tolerance) * 255)
            if alpha < 50:
                alpha = 0
            new_pixels[x, y] = (r, g, b, alpha)
        else:
            new_pixels[x, y] = (r, g, b, a)
            
        # Update bounding box if not fully transparent
        if new_pixels[x, y][3] > 10:
            if x < bbox[0]: bbox[0] = x
            if y < bbox[1]: bbox[1] = y
            if x > bbox[2]: bbox[2] = x
            if y > bbox[3]: bbox[3] = y

# Crop to bounding box
cropped_img = new_img.crop((bbox[0], bbox[1], bbox[2] + 1, bbox[3] + 1))

# Make it square by padding
cw, ch = cropped_img.size
size = max(cw, ch) + 20 # Add some padding

square_img = Image.new('RGBA', (size, size), (0, 0, 0, 0))
offset = ((size - cw) // 2, (size - ch) // 2)
